import 'package:flutter/material.dart';
import 'package:provider/provider.dart';

void main() {
  runApp(MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return ChangeNotifierProvider(
      create: (context) => MyAppState(),
      child: MaterialApp(
        title: 'Namer App',
        theme: ThemeData(
          colorScheme: ColorScheme.fromSeed(seedColor: const Color.fromARGB(255, 245, 186, 8)),
        ),
        home: MyHomePage(),
      ),
    );
  }
}

class MyAppState extends ChangeNotifier {

  // this function for applying radioButton values. don't change anything in this function
  void MysetState(int? value){
    radioButton = value;
    notifyListeners();
  }

int? radioButton = 0; /* linked with Radio Buttons .. buttons sum, sub and multiply have values 0, 1 and 2. 
Meaning if the user chooses sum, radioButton = 0 | if the user chooses sub, radioButton = 1 | if the user chooses multiply, radioButton = 2*/




  TextEditingController inputField1 = TextEditingController(); // linked with inputField1 .. use inputfield1.text for input value
  TextEditingController inputField2 = TextEditingController(); // linked with inputField2 .. use inputfield2.text for input value

  
  var textBox1 = 0; // linked with TextBox1



  // linked with calculate button
  void calculate(){


    notifyListeners();
  }

}

class MyHomePage extends StatefulWidget {
  @override
  State<MyHomePage> createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  
    var selectedIndex = 0; // create variable
  @override
  Widget build(BuildContext context) {

    var page;

    if (selectedIndex == 0){
      page = HomePage();
    }
    else if (selectedIndex == 1){
      page = ListPage();
    }

    return Scaffold(
      body: Row(
        children: [
          SafeArea(
            child: NavigationRail   (
              extended: false,
              destinations: [
                NavigationRailDestination(
                  icon: Icon(Icons.home),
                  label: Text('Home'),
                ),
                NavigationRailDestination(
                  icon: Icon(Icons.list),
                  label: Text('Favorites'),
                ),
              
              ],
              selectedIndex: selectedIndex,
              onDestinationSelected: (value) {
                setState(() {
                  selectedIndex = value;
                });;
              },
            ),
          ),
          Expanded(
            child: Container(
              color: Theme.of(context).colorScheme.primaryContainer,
              child: page,
            ),
          ),
        ],
      ),
    );
  }
}

class HomePage extends StatelessWidget {
  const HomePage({super.key});

  @override
  Widget build(BuildContext context) {
    var appState = context.watch<MyAppState>();
    

    return Center(
      child: Column(
        mainAxisAlignment: MainAxisAlignment.center,
        children: [
          RadioListTile<int>(
          title: const Text('sum  +'),
          value: 0,
          groupValue: appState.radioButton,
          onChanged: (int? value) {
             {
              appState.MysetState(value);
            };
          },
        ),
        // 3. Add second RadioListTile
        RadioListTile<int>(
          title: const Text('sub  -'),
          value: 1,
          groupValue: appState.radioButton,
          onChanged: (int? value) {
             {
              appState.MysetState(value);
            };
          },
        ),
        RadioListTile<int>(
          title: const Text('multiply   *'),
          value: 2,
          groupValue: appState.radioButton,
          onChanged: (int? value) {
             {
              appState.MysetState(value);
            };
          },
        ),
          SizedBox(height: 40),
          Row(
            mainAxisAlignment: MainAxisAlignment.center,
            children: [
              SizedBox(width: 200,child: TextField(controller: appState.inputField1, decoration: InputDecoration(border: OutlineInputBorder(), label: Text("inputField1")),)),
              SizedBox(width: 200,),
              SizedBox(width: 200,child: TextField(controller: appState.inputField2, decoration: InputDecoration(border: OutlineInputBorder(), label: Text("inputField2")),)),
            ],
          ),
          ElevatedButton(onPressed: appState.calculate, child: Text("calculate")),
          SizedBox(height: 40),
          Container(child: InputDecorator(decoration: InputDecoration(labelText: 'TextBox1',border: OutlineInputBorder(),),child: Text(appState.textBox1.toString()),), width: 200,), 
         Row(
            mainAxisSize: MainAxisSize.min,
            children: [
              SizedBox(width: 10), 
            ],
          ),
        ],
      ),
    );
  }
}
